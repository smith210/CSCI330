# SURLY COMMAND FILE CONTAINING THE DEMO FRUIT STAND

#RELATION
#DEFINITIONS
RELATION FRUIT (FNUM CHAR 3,TITLE CHAR 30, PRICE NUM 4);
RELATION STORE (FNUM CHAR 3,
                NAME CHAR 20);
RELATION SEASON (SNAME CHAR NAME 10, FNUM 3);

#INSERTING TUPLES
INSERT FRUIT A1 APPLE 3;
INSERT FRUIT B1 BANANA 1;
INSERT FRUIT G2 GRAPE 3;
INSERT FRUIT S8 STRAWBERRY 4;
INSERT FRUIT B2 BLUEBERRY 5;
INSERT FRUIT B3 BLACKBERRY 5;
INSERT FRUIT C1 RAINIER CHERRY 10;

INSERT STORE A1 'FRED MEYER';
INSERT STORE B1 'FRED MEYER';
INSERT STORE S8 'TRADER JOES';
INSERT STORE B3 'WHOLE FOODS';
INSERT STORE C1 'FARMERS MARKET';

INSERT SEASON SUMMER S8;
INSERT SEASON AUTUMN A1;

#PRINTING TUPLES
PRINT FRUIT, STORE;

#DELETE TUPLES
DELETE STORE;
PRINT FRUIT, STORE;
PRINT CATALOG;

#DESTROY RELATION
DESTROY STORE;
PRINT CATALOG, STORE;
